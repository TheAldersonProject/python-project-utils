File Handler Module

Overview

The File Handler module provides utilities for reading and parsing configuration files in various formats. It offers a standardized interface for working with TOML and YAML files, handling common error cases gracefully.

Key Features

  • Read file contents as text with proper encoding
  • Parse TOML files into Python dictionaries
  • Parse YAML files into Python dictionaries
  • Robust error handling with appropriate logging

Usage Examples

Reading a TOML File

from project_tools.core import file_handler

# Parse a TOML configuration file
config = file_handler.get_toml_file_content("pyproject.toml")
print(f"Project name: {config.get('project', {}).get('name')}")

Reading a YAML File

from project_tools.core import file_handler

# Parse a YAML configuration file
config = file_handler.get_yaml_file_content("config.yaml")
print(f"Database URL: {config.get('database', {}).get('url')}")

def file_content_as_text(filepath, encoding="utf-8"):
    """Read and return the content of a file as a string."""
    path = Path(filepath)
    return path.read_text(encoding=encoding)

def get_toml_file_content(filepath: str) -> dict[str, Any]:
    """Read and parse a TOML file, returning its content as a dictionary."""
    content = file_content_as_text(filepath)
    logging.debug("Toml content as text: \n%s\n", content)

    try:
        data: dict[str, Any] = tomllib.loads(content)
        logging.debug("Retrieve data: \n%s\n", data)
        return data
    except Exception as e:  # noqa
        logging.exception(e)
        logging.debug("Returning an empty dict.")
        return {}

def get_yaml_file_content(filepath: str) -> dict[str, Any]:
    """Read and parse a YAML file, returning its content as a dictionary."""
    content = file_content_as_text(filepath)
    logging.debug("Content data type %s", type(content))
    logging.debug("Yaml content as text: \n%s\n", content)

    try:
        data: dict[str, Any] = yaml.load(content, Loader=yaml.Loader)
        logging.debug("Retrieve data: \n%s\n", data)
        return data
    except Exception as e:  # noqa
        logging.exception(e)
        logging.debug("Returning an empty dict.")
        return {}
